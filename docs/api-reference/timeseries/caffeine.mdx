    apiKey: '<my_api_key>',
    environment: VitalEnvironment.Sandbox,
});

```java Java
import com.vital.api.Vital;
import com.vital.api.core.Environment;
import com.vital.api.resources.vitals.requests.VitalsCaffeineRequest;

Vital vital = Vital.builder()
        .apiKey("YOUR_API_KEY")
        .environment(Environment.SANDBOX)
        .build();

VitalsCaffeineRequest request = VitalsCaffeineRequest.builder()
        .startDate("2022-05-01")
        .endDate("2022-06-01")
        .build();

var data = vital.vitals().caffeine("<user_id>", request);
```

```go Go
import (
  "context"

  vital "github.com/tryVital/vital-go"
  vitalclient "github.com/tryVital/vital-go/client"
)

client := vitalclient.NewClient(
  vitalclient.WithApiKey("<YOUR_API_KEY>"),
  vitalclient.WithBaseURL(vital.Environments.Sandbox),
)

EndDate := "2022-06-01"
request := &vital.VitalsCaffeineRequest{
  StartDate: "2022-05-01",
  EndDate:   &EndDate,
}

response, err := client.Vitals.Caffeine(context.TODO(), "*", request)
if err != nil {
  return err
}
fmt.Printf("Received data %s\n", response)
```
